"--------------------
" 表示
"--------------------
" 行番号を表示
set number
" 対応する括弧()やブレース{}、ブラケット[]をハイライト
set showmatch matchtime=1
" 改行時などに自動でインデントを設定
set smartindent
" タブ幅をスペース2つ分にする
set tabstop=2
" 読み込み時などにtab幅をスペース2つ分に設定
set shiftwidth=2


"--------------------
" 検索
"--------------------
" 検索結果をハイライト
set hlsearch
" ファイル末尾まで検索した後、ファイル先頭から再び検索
set wrapscan
